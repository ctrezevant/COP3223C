# (c) 2017 Charlton Trezevant - MIT License

# Overview:
# 	This file will cause Make to automatically detect and build any C source files
# 	in the ./src/ directory whose name ends with `_main.c`.

# Usage:
#		To use this script,
#

AB_VERSION=1.0
PROJSRC=$(wildcard ${CURDIR}/src/*_main.c)
PROJNAME=$(subst _main.c,,$(subst ${CURDIR}/src/,,$(PROJSRC)))

all:$(PROJNAME)

$(PROJNAME):$(PROJSRC)
	$(info Make (Autobuild - Compile): Building "${PROJNAME}" from $^...)
	mkdir -p ${CURDIR}/bin/
	gcc -std=c99 -o ${CURDIR}/bin/$@ $^
	chmod +x ${CURDIR}/bin/$@
	@echo ""

test:$(PROJNAME)
ifneq ("$(wildcard ${CURDIR}/test_input.txt)","")

	@echo ""
else
	$(info Make (Autobuild - Test): Will not test "${PROJNAME}", as no test input is available! (${CURDIR}/test_input.txt doesn't exist))
	@echo ""
endif

clean:

	@echo ""

help:
